<using_directive> ::= "using" <identifier> <newline>
<include_directive> ::= "include" <identifier> <newline>
<statement> ::= <on_join_event> | <command> | <label_declaration> | <goto_statement> | <if_statement> | <label_call> | <comment> | <else_statement> | <package_unwrap>
<on_join_event> ::= "#onJoin" <newline> <statement>*
<command> ::= "msg" <string_literal> <newline>
            | "cpemsg" <identifier> <string_literal> <newline>
            | "localmsg" <identifier> <string_literal> <newline>
            | "menumsg" <identifier> <string_literal> <newline>
            | "delay" <identifier> <newline>
            | "jump" <identifier> <newline>
            | "goto" <identifier> <newline>
            | "call" <identifier> <newline>
            | "quit" <newline>
            | "terminate" <newline>
            | "newthread" <identifier> <newline>
            | "cmd" <external_command> <newline>
            | "set" <identifier> <identifier> <newline>
            | "setadd" <identifier> <identifier> <newline>
            | "setsub" <identifier> <identifier> <newline>
            | "setmul" <identifier> <identifier> <newline>
            | "setdiv" <identifier> <identifier> <newline>
            | "setmod" <identifier> <identifier> <newline>
            | "setpow" <identifier> <identifier> <newline>
            | "setsin" <identifier> <identifier> <newline>
            | "setcos" <identifier> <identifier> <newline>
            | "settan" <identifier> <identifier> <newline>
            | "setsqrt" <identifier> <identifier> <newline>
            | "setrandrange" <identifier> <identifier> <identifier> <newline>
            | "setrandrangedecimal" <identifier> <identifier> <identifier> <newline>
            | "setrandlist" <identifier> <identifier> <newline>
            | "setround" <identifier> <newline>
            | "setroundup" <identifier> <newline>
            | "setrounddown" <identifier> <newline>
            | "setfromhexcolor" <identifier> <identifier> <newline>
            | "settohexcolor" <identifier> <identifier> <newline>
            | "show" <identifier> <identifier>* <newline>
            | "kill" <string_literal>? <newline>
            | "resetdata" <identifier> <identifier>? <newline>
            | "item" <identifier> <identifier> <newline>
            | "freeze" <newline>
            | "unfreeze" <newline>
            | "look" <identifier> <newline>
            | "stare" <identifier>? <newline>
            | "env" <identifier> <identifier> <newline>
            | "motd" <identifier>* <newline>
            | "setspawn" <identifier> <newline>
            | "setdeathspawn" <identifier> <newline>
            | "allowmbrepeat" <newline>
            | "reply" <identifier> "|" <string_literal> "|" <identifier> <newline>
            | "reply" "clear" <newline>
            | "replysilent" <identifier> "|" <string_literal> "|" <identifier> <newline>
            | "replysilent" "clear" <newline>
            | "tempblock" <identifier> <newline>
            | "tempchunk" <identifier> <newline>
            | "boost" <identifier> <newline>
            | "effect" <identifier> <newline>
            | "reach" <identifier> <newline>
            | "setblockid" <identifier> <identifier> <newline>
            | "definehotkey" <identifier> "|" <identifier> "|" <identifier>* <newline>
            | "undefinehotkey" <identifier> "|" <identifier>* <newline>
            | "placeblock" <identifier> <identifier> <newline>
            | "changemodel" <identifier>? <newline>
            | "award" <identifier> <newline>
            | "setsplit" <identifier> <identifier>? <newline>
            | "setlength" <identifier> <identifier>? <newline>
            | "setdirvector" <identifier> <identifier> <identifier> <identifier> <identifier> <newline>
            | "clickevent" <identifier> "register" <identifier> <newline>
            | "clickevent" <identifier> "unregister" <newline>
            | "chatevent" "sync" "register" <identifier> <identifier> <string_literal> <newline>
            | "chatevent" "sync" "unregister" <newline>
            | "error" <string_literal> <newline>
            | "gui" <identifier> <identifier> <newline>
            | "gui" "reset" <newline>
<label_declaration> ::= <identifier> ":" <newline> <statement>*
<goto_statement> ::= "goto" <identifier> <newline>
<if_statement> ::= "if" <identifier> <command> <newline>
                  | "if" <identifier> "|=|" <identifier> <command> <newline>
                  | "if" <identifier> "|" <identifier> "|" <identifier> <command> <newline>
                  | "if" "item" <identifier> <command> <newline>
                  | "if" "label" <identifier> <command> <newline>
                  | "ifnot" <identifier> <command> <newline>
                  | "ifnot" <identifier> "|=|" <identifier> <command> <newline>
                  | "ifnot" <identifier> "|" <identifier> "|" <identifier> <command> <newline>
                  | "ifnot" "item" <identifier> <command> <newline>
                  | "ifnot" "label" <identifier> <command> <newline>
<else_statement> ::= "else" <command> <newline>
<label_call> ::= <identifier> "|" <arguments>* <newline>
<arguments> ::= <identifier> | <string_literal>
<comment> ::= "//" <string_literal> <newline>
<package_unwrap> ::= "{" <identifier> "}"
<string_literal> ::= <identifier>* | "\"" <identifier>* "\""
<external_command> ::= <identifier>*
<identifier> ::= [A-Za-z0-9_]+
<newline> ::= "\n"
<whitespace> ::= [ \t]+
